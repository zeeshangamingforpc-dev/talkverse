  <script>
    function speakText() {
      var text = document.getElementById('text').value;
      var lang = document.getElementById('voiceSelect').value;
      if(text !== "") {
        var utterance = new SpeechSynthesisUtterance(text);
        utterance.lang = lang;
        window.speechSynthesis.speak(utterance);
      } else {
        alert("Bhai pehle kuch type karo!");
      }
    }
  </script>
